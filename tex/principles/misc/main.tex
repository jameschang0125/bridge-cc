\subsection{cuebids}

\begin{itemize}
    \setlength\itemsep{0pt}
    \item if no agreement, cue shows strength and at in many situations GF
    \item show stop if two unbid suit or opp. bid two suits, and both cuebids are available. otherwise ask stop
    \item if one cuebid is unavailable, assume the stopper is good
\end{itemize}

\end{itemize}
