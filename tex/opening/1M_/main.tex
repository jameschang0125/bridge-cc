\begin{itemize}
    \setlength\itemsep{0pt}

    \item 1S - 3H \& 1H - 2N = limit raise, 4+M inv
    \item 1M - 3M = pre
    \item 1M - 3N = (4333), CoG
    \item double jump = spl, roughly GF minimum
    \item stronger splinters can use auto-splinter after 2-over-1
\end{itemize}
